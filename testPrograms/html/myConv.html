convolution function

CNspikeToCurrentTau=0.01;
t=dt:dt:3*CNspikeToCurrentTau;
CNalphaFunction=...
    (1/CNspikeToCurrentTau)*t.*exp(-t/CNspikeToCurrentTau);

plot(t, CNalphaFunction)
